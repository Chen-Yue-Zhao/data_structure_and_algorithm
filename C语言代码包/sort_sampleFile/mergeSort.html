二路归并算法

　

void mergeSort(SortObject  *pvector)

{

    RecordNode record[MAXNUM];

    int length;

    length=1;

    while(length<pvector->n)

    {

        mergePass(pvector->record,record,pvector->n,length);

                    /*一趟归并,结果存放在数组r1中*/

        length*=2;

        mergePass(record,pvector->record,pvector->n,length);

                        /*一趟归并,结果存放在数组r中*/

        length*=2;

    }

}
