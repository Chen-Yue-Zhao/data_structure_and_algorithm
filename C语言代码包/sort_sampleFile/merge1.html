两组归并算法

　

　

　

void merge(RecordNode r[ ],RecordNode r1[ ],int low,int m,int high)

{                       /*r[low]到r[m]和r[m+1]到r[high]是两个有序文件*/

    int i,j,k;

    i=low;j=m+1;k=low;

    while((i<=m)&&(j<=high))

    { 

         if(r[i].key<=r[j].key)  

                        /*从两个有序文件中的第一个记录中选出小的记录*/

         r1[k++]=r[i++];

      else

        r1[k++]=r[j++];

    }

    while(i<=m) r1[k++]=r[i++];        /*复制第一个文件的剩余记录*/

    while(j<=high) r1[k++]=r[j++];     /*复制第二个文件的剩余记录*/

}
