中序遍历二叉树非递归算法


/*******************************************************************

思想：
从根结点开始，令变量p为根结点，若p不为空结点，则令p沿其左子女结点前进，在前进过程中，把所经过的结点逐个压入栈中。当p为空结点，从栈中弹出栈顶元素赋给变量p，并访问该结点，令p为它当前指向结点的右子女结点。重复上述过程，当p为空时并且栈也为空时，周游结束。

　

　

　

struct BinTreeNode    /* 二叉树中结点 */
typedef struct BinTreeNode *PBinTreeNode;   /*结点的指针类型*/
struct BinTreeNode
{ 

     DataType info;    /* 数据域 */
    PBinTreeNode llink;   /* 指向左子女 */
    PBinTreeNode rlink;   /* 指向右子女 */
};
typedef struct BinTreeNode *BinTree;    /*二叉树定义*/
typedef BinTree *PBinTRee;  /* 二叉树类型的指针类型 */

　

　

　

#define MAXNUM 1000    /* 栈中最大元素个数 */
struct SeqStack    /*顺序栈类型定义*/
{ 
    P BinTreeNode s[MAXNUM];
    int top;   /* 指示栈顶位置，空栈=-1 */
}
typedef struct SeqStack, *PSeqStack;    /*顺序栈类型的指针类型*/

　

void npreOrder(PBinTreeNode t)
{

    PSeqStack st;
   PBinTreeNode p;     //栈中存放各个左子树根结点
     p = t;
   do 
   { 
        while (p != NULL)
       {
            push(st,p->llink);

            p = p->llink;   //继续搜索p的左子树 
       }   

         p=top_seq(st);
       pop(st);     //左子树根结点出栈，转到搜索右子树

             visit(p);

        p=p->rlink;    

     } while (!isEmptyStack(st));
}
