void inOrder(Node p)

{ 

   Node c;

    c = leftChild(q);                //获取该结点的长子

   if (c==NULL)  

        visit(p);                //不存在，访问该结点

  else

   {  

       inOrder(c);                 //中根遍历长子

       visit(p);                      //访问根

       c = rightSibling(c);    //中根遍历其它子女

       while (c!=NULL)

       {   

            inorder(c);

            c = rightSibling(c);

       }

    }

}
