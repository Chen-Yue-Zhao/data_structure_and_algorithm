void postOrder(Node p)

{  Node c;

    c = leftChild(q);          //获取该结点的长子

        //然后按照从左往右的次序后根遍历该结点的所有子女

     while (c!=NULL)

     { 

          postOrder( c);                //后根遍历

         c = rightSibling(c);        //继续后根遍历右兄弟

     }

}
